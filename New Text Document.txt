CREATE TABLE CUSTOMER(CUST_ID VARCHAR(50) PRIMARY KEY, CUST_NAME VARCHAR(40) NOT NULL, CUST_PHONE BIGINT UNIQUE NOT NULL, CUST_ADDRESS VARCHAR(125) UNIQUE NOT NULL, CUST_PINCODE MEDIUMINT NOT NULL, CUST_PASSWORD VARCHAR(20) NOT NULL, CUST_GENDER CHAR(1), CUST_DOB DATE);    

CREATE TABLE SELLER(SELLER_ID VARCHAR(50), SELLER_GST VARCHAR(15), SELLER_PHONE BIGINT UNIQUE NOT NULL, SELLER_ADDRESS VARCHAR(125) NOT NULL UNIQUE, SELLER_PINCODE MEDIUMINT NOT NULL, SELLER_PASSWORD VARCHAR(20) NOT NULL, NUM_ORDERS INT, PRIMARY KEY(SELLER_ID, SELLER_GST)); 

CREATE TABLE ITEMS(ITEM_ID VARCHAR(15) PRIMARY KEY, ITEM_NAME VARCHAR(20) NOT NULL, ITEM_CATEGORY VARCHAR(10) NOT NULL, ITEM_PRICE SMALLINT NOT NULL); 

CREATE TABLE ORDERS(ORDER_ID VARCHAR(15) PRIMARY KEY, ORDER_STATUS VARCHAR(15) NOT NULL, ITEM_ID VARCHAR(15) , CUST_ID VARCHAR(50) , CUST_PHONE BIGINT UNIQUE NOT NULL, CUST_ADDRESS VARCHAR(125) UNIQUE NOT NULL , SELLER_ID VARCHAR(50),
FOREIGN KEY (ITEM_ID) REFERENCES ITEMS(ITEM_ID),
FOREIGN KEY (CUST_ID) REFERENCES CUSTOMER (CUST_ID),
FOREIGN KEY (CUST_ADDRESS) REFERENCES CUSTOMER (CUST_ADDRESS),
FOREIGN KEY (CUST_PHONE) REFERENCES CUSTOMER (CUST_PHONE),
FOREIGN KEY (SELLER_ID) REFERENCES SELLER (SELLER_ID) );
